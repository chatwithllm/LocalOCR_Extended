> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a `/features` page serving a sidebar+detail-pane visual reference for all 30 features of LocalOCR Extended.

**Architecture:** Separate Flask blueprint (`handle_features.py`) serves `features.html` (public, no auth) and `features-data.js` (auth-guarded). Self-contained page with own CSS, sidebar search+groups, JS-rendered detail pane. Feature data is `window.FEATURES_DATA` array generated by stub script then manually enriched.

**Tech Stack:** Flask (Blueprint, send_from_directory), vanilla JS (no framework), inline CSS (dark theme, self-contained), Python stub generator script.

  // safeSetHTML uses createContextualFragment to parse HTML through the
  // browser's HTML parser rather than direct property assignment.
  // All content comes from developer-authored static data served via an
  // auth-guarded endpoint (/features/data) — not from user input.
  function safeSetHTML(el, html) {
    el.textContent = '';
    const range = document.createRange();
    range.selectNode(el);
    el.appendChild(range.createContextualFragment(html));
  }

  (async () => {
    try {
      const res = await fetch('/features/data');
      if (res.status === 401) { window.location.href = '/'; return; }
      if (!res.ok) {
        document.getElementById('detail-empty').textContent = 'Failed to load feature data.';
        return;
      }
      const text = await res.text();
      const script = document.createElement('script');
      script.textContent = text;
      document.head.appendChild(script);
      initFeatures();
    } catch (e) {
      document.getElementById('detail-empty').textContent = 'Error: ' + e.message;
    }
  })();

  let _activeId = null;

  function initFeatures() {
    if (!window.FEATURES_DATA || !window.FEATURES_DATA.length) {
      document.getElementById('detail-empty').textContent = 'No feature data loaded.';
      return;
    }
    buildSidebar();
    const first = window.FEATURES_DATA[0]?.features?.[0];
    if (first) selectFeature(first.id);
  }

  function buildSidebar(filter) {
    const list = document.getElementById('sidebar-list');
    const q = (filter || '').toLowerCase();
    let html = '';
    for (const group of window.FEATURES_DATA) {
      const anyVisible = !q || group.features.some(f =>
        f.title.toLowerCase().includes(q) || (f.tagline || '').toLowerCase().includes(q));
      if (!anyVisible) continue;
      html += '<div class="group-label">' + group.label + '</div>';
      for (const f of group.features) {
        const match = !q || f.title.toLowerCase().includes(q) || (f.tagline || '').toLowerCase().includes(q);
        const cls = ['feat-item', f.id === _activeId ? 'active' : '', match ? '' : 'hidden'].filter(Boolean).join(' ');
        html += '<div class="' + cls + '" data-id="' + f.id + '" onclick="selectFeature(\'' + f.id + '\')">'
              + f.icon + ' ' + f.title + '</div>';
      }
    }
    safeSetHTML(list, html);
  }

  function filterFeatures(q) { buildSidebar(q); }

  function selectFeature(id) {
    _activeId = id;
    let feat = null;
    for (const group of window.FEATURES_DATA) {
      feat = group.features.find(f => f.id === id);
      if (feat) break;
    }
    if (!feat) return;
    buildSidebar(document.getElementById('search').value);
    renderDetail(feat);
  }

  function renderDetail(f) {
    document.getElementById('detail-empty').style.display = 'none';
    const el = document.getElementById('detail-content');
    el.style.display = 'block';

    const badgeMap = { Web: 'badge-web', Mobile: 'badge-mobile', Telegram: 'badge-telegram' };
    const badges = (f.platforms || [])
      .map(p => '<span class="badge ' + (badgeMap[p] || 'badge-web') + '">' + p + '</span>').join('');

    const whereHtml = f.where
      ? f.where.split('&#8594;').join('→').split('→').map((s, i) =>
          i === 0
            ? '<span class="where-icon">&#128205;</span><span style="color:var(--muted)">' + s.trim() + '</span>'
            : '<span style="color:var(--muted2)">&#8594;</span><span class="where-crumb">' + s.trim() + '</span>'
        ).join('')
      : '';

    const steps = f.flow || [];
    const flowHtml = steps.map((step, i) => {
      const cls = i === 0 ? 'flow-step--start' : i === steps.length - 1 ? 'flow-step--end' : 'flow-step--mid';
      const arrow = i < steps.length - 1 ? '<span class="flow-arrow">&#8594;</span>' : '';
      return '<div class="flow-step ' + cls + '">'
           + '<div class="flow-step-icon">' + step.icon + '</div>'
           + '<div class="flow-step-label">' + step.label + '</div>'
           + (step.sub ? '<div class="flow-step-sub">' + step.sub + '</div>' : '')
           + '</div>' + arrow;
    }).join('');

    const interactionsHtml = (f.interactions || [])
      .map(item => '<div class="interaction-item">' + item + '</div>').join('');

    const tipHtml = f.tip
      ? '<div class="tip-box"><span class="tip-icon">&#128161;</span><span>' + f.tip + '</span></div>' : '';

    const html = [
      '<div class="feat-header">',
        '<div>',
          '<div class="feat-title"><span class="feat-icon">' + f.icon + '</span>' + f.title + '</div>',
          '<div class="feat-tagline">' + (f.tagline || '') + '</div>',
        '</div>',
        '<div class="badges">' + badges + '</div>',
      '</div>',
      whereHtml ? '<div class="where-box">' + whereHtml + '</div>' : '',
      flowHtml  ? '<div class="section-label">How it works</div><div class="flow">' + flowHtml + '</div>' : '',
      f.mockup  ? '<div class="section-label">What you see</div><div class="mockup-wrap">' + f.mockup + '</div>' : '',
      interactionsHtml ? '<div class="section-label">Key interactions</div><div class="interactions-grid">' + interactionsHtml + '</div>' : '',
      tipHtml,
    ].join('');

    safeSetHTML(el, html);
    document.getElementById('detail').scrollTop = 0;
  }
</script>

def read_specs():
    taglines = {}
    if not os.path.isdir(SPECS_DIR):
        return taglines
    for fname in os.listdir(SPECS_DIR):
        if not fname.endswith(".md"):
            continue
        with open(os.path.join(SPECS_DIR, fname)) as fh:
            text = fh.read()
        m = re.search(r'##\s+Goal\s*\n+(.*?)(\n\n|\n##)', text, re.DOTALL)
        if not m:
            continue
        tagline = m.group(1).strip().split("\n")[0].strip("- ")
        lower = fname.lower().replace("-", "").replace("_", "")
        for fid in FEATURE_META:
            if fid.replace("-", "") in lower:
                taglines[fid] = tagline
                break
    return taglines

if __name__ == "__main__":
